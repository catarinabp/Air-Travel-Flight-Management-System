134 if(endAirport != startAirport) {
135 // Map to store distances from startAirport to each vertex
136 unordered_map<T, int> distance;
137
138 // Priority queue to store vertices and their distances
139 priority_queue<pair<int, T>, vector<pair<int, T>>, greater<pair<int, T>>> pq;
140
141 // Initialize distances
142 for (const auto &vertex : graph.getVertexSet()) {
143 if(!vertex->getAdj().empty()) {
144 if(vertex->getAdj()[0].getWeight() == "AIRPORT") {
145 distance[vertex->getInfo()] = INT_MAX;
146 }
147 }
148 }
149
150 // Set distance for the startAirport to 0
151 distance[startAirport] = 0;
152
153 // Insert startAirport into the priority queue
154 pq.push({0, startAirport});
155
156 // Dijkstra's algorithm
157 while (!pq.empty()) {
158 // Extract the vertex with the smallest distance
159 T currentAirport = pq.top().second;
160 pq.pop();
161
162 // Get neighboring vertices of the currentAirport
163 if(!graph.findVertex(currentAirport)->getAdj().empty()) {
164 auto neighbors = graph.findVertex(currentAirport)->getAdj();
165 if(neighbors[0].getWeight() == "AIRPORT") {
166 neighbors.erase(neighbors.begin());
167 }
168 // Update distances to neighboring vertices
169 for (const auto &edge : neighbors) {
170 T neighborAirport = edge.getDest()->getInfo();
171 int newDistance = distance[currentAirport] + 1; // Assuming equal weight for all edges
172
173 // If a shorter path is found, update the distance
174 if (newDistance < distance[neighborAirport] || distance[neighborAirport] == INT_MAX) {
175 distance[neighborAirport] = newDistance;
176 pq.push({newDistance, neighborAirport});
177 }
178 }
179 }
180 }
181
182 // Reconstruct all paths from endAirport to startAirport
183 backtrackPaths(graph,distance, endAirport, startAirport, currentPath, allPaths);
184 }
185
186 return allPaths;
